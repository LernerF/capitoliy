#include <stdio.h>
#include <stdbool.h>

// Функция для вычисления модуля разности двух чисел
int abs_diff(int a, int b) {
    return (a > b) ? (a - b) : (b - a);
}

int main() {
    int num;
    
    while (scanf("%d", &num) == 1) {
        int digit;
        int prev_digit = -1;
        bool result = true;
        
        while (num > 0) {
            digit = num % 10;
            
            if (prev_digit != -1) {
                switch(abs_diff(prev_digit, digit)) {
                    case 0:
                        result = false;
                        break;
                    default:
                        break;
                }
            }
            
            prev_digit = digit;
            num /= 10;
        }
        
        printf("%d\n", result);
    }
    
    return 0;
}
