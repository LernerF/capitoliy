```
#include <stdio.h>
#include <string.h>

int main() {
    char word[100];
    printf("Enter a word: ");
    scanf("%s", word);

    int len = strlen(word);
    int i;
    int vowels[5] = {0}; // a, e, i, o, u

    for (i = 0; i < len; i++) {
        char c = word[i];
        if (c == 'a' || c == 'A') {
            vowels[0] = 1;
        } else if (c == 'e' || c == 'E') {
            vowels[1] = 1;
        } else if (c == 'i' || c == 'I') {
            vowels[2] = 1;
        } else if (c == 'o' || c == 'O') {
            vowels[3] = 1;
        } else if (c == 'u' || c == 'U') {
            vowels[4] = 1;
        }
    }

    int distinct_vowels = 0;
    for (i = 0; i < 5; i++) {
        if (vowels[i] == 1) {
            distinct_vowels++;
        }
    }

    if (distinct_vowels == 5) {
        printf("Все гласные различны.\n");
    } else {
        printf("Все гласные не различны.\n");
    }

    return 0;
}
```
