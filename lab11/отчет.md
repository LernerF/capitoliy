6. **Протокол**:
```
#include <stdio.h>
#include <stdlib.h>

int main() {
  float celsius, fahrenheit;
  int result;

  printf("Введите температуры в градусах Цельсия (для завершения введите нечисловое значение):\n");

  while (1) {
    // Чтение ввода
    result = scanf("%f", &celsius);

    // Проверка на числовой ввод
    if (result != 1) {
      // Ввод не является числом, прерываем цикл
      break;
    }

    // Перевод температуры из Цельсия в Фаренгейт
    fahrenheit = (celsius * 9/5) + 32;

    // Вывод результата
    printf("Температура в градусах Фаренгейта: %.2f\n", fahrenheit);
  }

  return 0;
}
```
